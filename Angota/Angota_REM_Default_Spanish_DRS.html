    Autores
    • Rafael Gómez García

    Descripción PD
    Parámetros Ninguno
    Expresiones de precondición Exista el IdComercial: Personal.allInstances() -> one( p | p.idComercial = idComercial and p.oclIsTypeOf(Comercial) = true)

    Expresiones de precondición (OCL) Exista el IdComercial: true
    Expresiones de postcondición Añadidos datos de comercial: let
    comer : Comercial = Personal.allInstances() -> any( p | p.idComercial = idComercial and p.oclIsTypeOf(Comercial) = true)

    in
    comer.nombre = nombre and
    comer.DNI = DNI and
    comer.direccion = direccion and
    comer.telefono = telefono and
    comer.email = email and
    comer.comision = comision
